# ## 2. Utility Function

# %% [code]
def utility(selected_power, true_power, hyperopia_penalty=2.0):
    """
    Calculate the utility of selecting a given IOL power.
    Hyperopia (selected < true) is penalized more heavily than myopia.
    """
    error = selected_power - true_power  # positive = myopic, negative = hyperopic
    if error < 0:
        return -hyperopia_penalty * abs(error)
    else:
        return -1.0 * abs(error)

def get_log_likelihood(candidate, predictions, sigmas):
    """Compute log-likelihood of a candidate power given formula predictions."""
    return np.sum(norm.logpdf(candidate, loc=predictions, scale=sigmas))

def get_correlated_log_likelihood(candidate, predictions, sigmas, rho=0.4):
    """
    Compute log-likelihood with correlated errors.
    Uses multivariate normal with covariance matrix:
    Cov(X_i, X_j) = rho * sigma_i * sigma_j (off-diagonal)
    Cov(X_i, X_i) = sigma_i^2 (diagonal)
    """
    n = len(predictions)
    # Build covariance matrix with pairwise correlation rho
    cov = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j:
                cov[i, j] = sigmas[i] ** 2
            else:
                cov[i, j] = rho * sigmas[i] * sigmas[j]
    
    diff = candidate - predictions
    try:
        inv_cov = np.linalg.pinv(cov)
        log_lik = -0.5 * diff @ inv_cov @ diff.T
        return log_lik
    except:
        # Fallback to independent if matrix is singular
        return np.sum(norm.logpdf(candidate, loc=predictions, scale=sigmas))

# ## 3. Core Simulation Function

# %% [code]
def run_simulation(trigger_threshold=0.50, hyperopia_penalty=2.0, correlated=False, verbose=True):
    """
    Run the Monte Carlo simulation with specified parameters.
    
    For correlated errors (rho = CORRELATION_RHO):
        X_i = μ + σ_i * (sqrt(rho) * Z + sqrt(1-rho) * ε_i)
        where Z ~ N(0,1) and ε_i ~ N(0,1) independent
        This yields Var(X_i) = σ_i^2 and Corr(X_i, X_j) = rho
    """
    start_time = time.time()
    
    # Step 1: Generate true powers
    true_powers = np.random.uniform(10.0, 30.0, N_SIMS)
    
    # Step 2: Generate formula predictions
    preds = np.zeros((N_SIMS, n_formulas))
    if correlated:
        # Shared component: Z ~ N(0,1) scaled by sqrt(rho)
        shared_noise = np.random.normal(0, 1, N_SIMS)  # Standard normal Z
        for i, sd in enumerate(sigma_values):
            # Independent component: ε_i ~ N(0,1) scaled by sqrt(1-rho)
            independent_noise = np.random.normal(0, 1, N_SIMS)
            # X_i = μ + σ_i * (sqrt(rho) * Z + sqrt(1-rho) * ε_i)
            preds[:, i] = true_powers + sd * (np.sqrt(CORRELATION_RHO) * shared_noise + 
                                               np.sqrt(1 - CORRELATION_RHO) * independent_noise)
    else:
        for i, sd in enumerate(sigma_values):
            preds[:, i] = np.random.normal(loc=true_powers, scale=sd, size=N_SIMS)
    
    # Step 3: Identify triggered cases (range >= trigger_threshold)
    ranges = np.max(preds, axis=1) - np.min(preds, axis=1)
    triggered = ranges >= trigger_threshold
    triggered_indices = np.where(triggered)[0]
    n_triggered = len(triggered_indices)
    
    if verbose:
        print(f"📊 Simulation completed in {time.time() - start_time:.2f} seconds")
        print(f"  • Total eyes simulated: {N_SIMS:,}")
        print(f"  • Triggered cases (range ≥ {trigger_threshold} D): {n_triggered:,} ({n_triggered/N_SIMS*100:.1f}%)")
        print(f"  • Consensus cases: {N_SIMS - n_triggered:,} ({(N_SIMS - n_triggered)/N_SIMS*100:.1f}%)")
    
    # Step 4: Initialize result arrays for triggered cases
    n = n_triggered
    
    # Initialize dictionaries for all strategies
    strategies = {
        'bayes': {'errors': np.zeros(n), 'surprise': np.zeros(n, dtype=bool), 
                  'hyperopic': np.zeros(n, dtype=bool), 'utility': np.zeros(n)},
        'mle': {'errors': np.zeros(n), 'surprise': np.zeros(n, dtype=bool), 
                'hyperopic': np.zeros(n, dtype=bool), 'utility': np.zeros(n)},
        'random': {'errors': np.zeros(n), 'surprise': np.zeros(n, dtype=bool), 
                   'hyperopic': np.zeros(n, dtype=bool), 'utility': np.zeros(n)},
        'myopic': {'errors': np.zeros(n), 'surprise': np.zeros(n, dtype=bool), 
                   'hyperopic': np.zeros(n, dtype=bool), 'utility': np.zeros(n)},
        'median': {'errors': np.zeros(n), 'surprise': np.zeros(n, dtype=bool), 
                   'hyperopic': np.zeros(n, dtype=bool), 'utility': np.zeros(n)},
        'weighted': {'errors': np.zeros(n), 'surprise': np.zeros(n, dtype=bool), 
                     'hyperopic': np.zeros(n, dtype=bool), 'utility': np.zeros(n)}
    }
    
    if verbose:
        print(f"\n🔄 Running Bayesian decision framework on triggered cases...")
        start_loop = time.time()
    
    for idx, eye_idx in enumerate(triggered_indices):
        true_p = true_powers[eye_idx]
        preds_eye = preds[eye_idx, :]
        
        # Candidate powers (unique 0.50 D steps)
        candidates = np.unique(np.round(preds_eye * 2) / 2)
        n_candidates = len(candidates)
        
        # Calculate log-likelihoods for all candidates
        log_likelihoods = np.zeros(n_candidates)
        for j, cand in enumerate(candidates):
            if correlated:
                log_likelihoods[j] = get_correlated_log_likelihood(cand, preds_eye, sigma_values, CORRELATION_RHO)
            else:
                log_likelihoods[j] = get_log_likelihood(cand, preds_eye, sigma_values)
        
        # Posterior probabilities (softmax of log-likelihoods)
        max_ll = np.max(log_likelihoods)
        likelihoods = np.exp(log_likelihoods - max_ll)
        posterior_probs = likelihoods / np.sum(likelihoods)
        
        # --- Bayesian Decision: Maximize Expected Utility ---
        expected_utilities = np.zeros(n_candidates)
        for j, cand in enumerate(candidates):
            exp_util = 0.0
            for k, other_cand in enumerate(candidates):
                exp_util += posterior_probs[k] * utility(cand, other_cand, hyperopia_penalty)
            expected_utilities[j] = exp_util
        
        bayes_power = candidates[np.argmax(expected_utilities)]
        
        # --- Pure MLE (max log-likelihood) ---
        mle_power = candidates[np.argmax(log_likelihoods)]
        
        # --- Median Consensus ---
        median_power = np.median(candidates)
        median_power = np.round(median_power * 2) / 2
        
        # --- Weighted Consensus (precision-weighted average) ---
        weights = 1.0 / (sigma_values ** 2)
        weighted_mean = np.sum(preds_eye * weights) / np.sum(weights)
        weighted_power = np.round(weighted_mean * 2) / 2
        
        # --- Random Selection ---
        random_power = np.random.choice(candidates)
        
        # --- Always-Myopic ---
        myopic_power = np.max(candidates)
        
        # Store all strategy selections and calculate metrics
        strategy_powers = {
            'bayes': bayes_power,
            'mle': mle_power,
            'random': random_power,
            'myopic': myopic_power,
            'median': median_power,
            'weighted': weighted_power
        }
        
        for name, selected_power in strategy_powers.items():
            error = abs(selected_power - true_p)
            strategies[name]['errors'][idx] = error
            strategies[name]['surprise'][idx] = error > 0.50
            strategies[name]['hyperopic'][idx] = selected_power < true_p
            strategies[name]['utility'][idx] = utility(selected_power, true_p, hyperopia_penalty)
    
    if verbose:
        print(f"✅ Decision framework applied to {n:,} triggered cases in {time.time() - start_loop:.2f} seconds")
    
    return strategies, true_powers, preds, triggered_indices

# %% [code]
def compute_metrics(strategies, n_triggered):
    """Compute aggregate metrics for all strategies."""
    results = {}
    strategy_names = list(strategies.keys())
    
    for name in strategy_names:
        errors = strategies[name]['errors']
        results[name] = {
            'mae': np.mean(errors),
            'std': np.std(errors),
            'surprise_pct': np.mean(strategies[name]['surprise']) * 100,
            'hyperopic_pct': np.mean(strategies[name]['hyperopic']) * 100,
            'myopic_pct': (1 - np.mean(strategies[name]['hyperopic'])) * 100,
            'utility': np.mean(strategies[name]['utility'])
        }
    
    return pd.DataFrame(results).T

# Compute metrics
df_results = compute_metrics(strategies, n_triggered_main)

# Label mapping

    'bayes': 'Bayesian Decision',
    'mle': 'Pure MLE',
    'random': 'Random Selection',
    'myopic': 'Always-Myopic',
    'median': 'Median Consensus',
    'weighted': 'Weighted Consensus'
}

df_results.index = [label_map.get(idx, idx) for idx in df_results.index]
