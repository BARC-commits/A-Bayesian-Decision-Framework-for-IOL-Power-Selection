# # Monte Carlo Simulation: Bayesian Decision Framework for IOL Power Selection
# **N = 100,000 Virtual Eyes | Computational Proof-of-Concept**
# 
# This notebook implements a Bayesian decision-theoretic framework to resolve IOL power disagreements among six modern formulas (Cooke K6, EVO, Hill-RBF 3.0, Hoffer QST, Kane, Pearl-DGS). The framework balances formula consensus (likelihood) with asymmetric clinical risk (utility) to select the IOL power that maximizes expected utility.
# 
# ---

# ## 1. Simulation Parameters

# %% [code]
# --- Simulation size ---
N_SIMS = 100000  # 100,000 virtual eyes
np.random.seed(42)  # Reproducibility

# --- Formula SDs from ESCRS validation study (after zero-centering) ---
FORMULAS = {
    'Cooke_K6': 0.44,
    'EVO': 0.47,
    'Hill_RBF': 0.48,
    'Hoffer_QST': 0.50,
    'Kane': 0.48,
    'Pearl_DGS': 0.46
}

formula_names = list(FORMULAS.keys())
sigma_values = np.array(list(FORMULAS.values()))
n_formulas = len(formula_names)

# --- Utility function parameters ---
HYPEROPIA_PENALTY = 2.0  # Hyperopia is twice as undesirable as myopia
MYOPIA_PENALTY = 1.0

# --- Trigger condition (range ≥ 0.75 D yields ~87% trigger rate) ---
TRIGGER_RANGE = 0.75  # D

print("=" * 60)
print("SIMULATION PARAMETERS (BAYESIAN DECISION THEORY)")
print("=" * 60)
print(f"Number of virtual eyes: {N_SIMS:,}")
print(f"Trigger condition (range ≥): {TRIGGER_RANGE} D")
print(f"Hyperopia penalty: {HYPEROPIA_PENALTY}x worse than myopia")
print(f"Myopia penalty: {MYOPIA_PENALTY}x")
print("\nFormula Standard Deviations (prediction error):")
for name, sd in FORMULAS.items():
    print(f"  • {name}: {sd:.2f} D")
print("=" * 60)

# %% [markdown]
# ## 2. Utility Function

# %% [code]
def utility(selected_power, true_power):
    """
    Calculate the utility of selecting a given IOL power.
    Hyperopia (selected < true) is penalized more heavily than myopia.
    """
    error = selected_power - true_power  # positive = myopic, negative = hyperopic
    if error < 0:
        return -HYPEROPIA_PENALTY * abs(error)
    else:
        return -MYOPIA_PENALTY * abs(error)

def get_log_likelihood(candidate, predictions, sigmas):
    """Compute log-likelihood of a candidate power given formula predictions."""
    return np.sum(norm.logpdf(candidate, loc=predictions, scale=sigmas))

# Test the utility function
print("\nUtility function test:")
true_p = 20.0
print(f"  True power: {true_p:.2f} D")
print(f"  Select 19.50 D (hyperopic by 0.50 D): Utility = {utility(19.50, true_p):.2f}")
print(f"  Select 20.50 D (myopic by 0.50 D): Utility = {utility(20.50, true_p):.2f}")
print(f"  Select {true_p:.2f} D (perfect): Utility = {utility(true_p, true_p):.2f}")
print(f"  Hyperopia penalty: {HYPEROPIA_PENALTY}x worse than myopia\n")

# %% [markdown]
# ## 3. Run the Monte Carlo Simulation (N = 100,000)

# %% [code]
start_time = time.time()

# Step 1: Generate true powers (uniform distribution across clinical range)
true_powers = np.random.uniform(10.0, 30.0, N_SIMS)

# Step 2: Generate formula predictions with independent Gaussian noise
preds = np.zeros((N_SIMS, n_formulas))
for i, sd in enumerate(sigma_values):
    preds[:, i] = np.random.normal(loc=true_powers, scale=sd, size=N_SIMS)

# Step 3: Identify triggered cases (range ≥ TRIGGER_RANGE)
ranges = np.max(preds, axis=1) - np.min(preds, axis=1)
triggered = ranges >= TRIGGER_RANGE
triggered_indices = np.where(triggered)[0]
n_triggered = len(triggered_indices)

print(f"📊 Simulation completed in {time.time() - start_time:.2f} seconds")
print(f"  • Total eyes simulated: {N_SIMS:,}")
print(f"  • Triggered cases (range ≥ {TRIGGER_RANGE} D): {n_triggered:,} ({n_triggered/N_SIMS*100:.1f}%)")
print(f"  • Consensus cases: {N_SIMS - n_triggered:,} ({(N_SIMS - n_triggered)/N_SIMS*100:.1f}%)\n")

# Step 4: Initialize result arrays for triggered cases
n = n_triggered
bayes_errors = np.zeros(n)
random_errors = np.zeros(n)
myopic_errors = np.zeros(n)
mle_errors = np.zeros(n)

bayes_surprise = np.zeros(n, dtype=bool)
random_surprise = np.zeros(n, dtype=bool)
myopic_surprise = np.zeros(n, dtype=bool)
mle_surprise = np.zeros(n, dtype=bool)

bayes_hyperopic = np.zeros(n, dtype=bool)
random_hyperopic = np.zeros(n, dtype=bool)
myopic_hyperopic = np.zeros(n, dtype=bool)
mle_hyperopic = np.zeros(n, dtype=bool)

bayes_utility = np.zeros(n)
random_utility = np.zeros(n)
myopic_utility = np.zeros(n)
mle_utility = np.zeros(n)

# Step 5: Loop over triggered cases and apply decision framework
print("🔄 Running Bayesian decision framework on triggered cases...")
start_loop = time.time()

for idx, eye_idx in enumerate(triggered_indices):
    true_p = true_powers[eye_idx]
    preds_eye = preds[eye_idx, :]
    
    # Candidate powers (unique 0.50 D steps)
    candidates = np.unique(np.round(preds_eye * 2) / 2)
    n_candidates = len(candidates)
    
    # Calculate log-likelihoods for all candidates
    log_likelihoods = np.zeros(n_candidates)
    for j, cand in enumerate(candidates):
        log_likelihoods[j] = get_log_likelihood(cand, preds_eye, sigma_values)
    
    # Posterior probabilities (softmax of log-likelihoods)
    max_ll = np.max(log_likelihoods)
    likelihoods = np.exp(log_likelihoods - max_ll)
    posterior_probs = likelihoods / np.sum(likelihoods)
    
    # Expected utility for each candidate
    expected_utilities = np.zeros(n_candidates)
    for j, cand in enumerate(candidates):
        exp_util = 0.0
        for k, other_cand in enumerate(candidates):
            exp_util += posterior_probs[k] * utility(cand, other_cand)
        expected_utilities[j] = exp_util
    
    # Bayesian Decision: select candidate with max expected utility
    bayes_power = candidates[np.argmax(expected_utilities)]
    
    # Pure MLE (max log-likelihood)
    mle_power = candidates[np.argmax(log_likelihoods)]
    
    # Comparator strategies
    random_power = np.random.choice(candidates)
    myopic_power = np.max(candidates)
    
    # Errors
    bayes_errors[idx] = abs(bayes_power - true_p)
    random_errors[idx] = abs(random_power - true_p)
    myopic_errors[idx] = abs(myopic_power - true_p)
    mle_errors[idx] = abs(mle_power - true_p)
    
    # Surprise (error > 0.50 D)
    bayes_surprise[idx] = bayes_errors[idx] > 0.50
    random_surprise[idx] = random_errors[idx] > 0.50
    myopic_surprise[idx] = myopic_errors[idx] > 0.50
    mle_surprise[idx] = mle_errors[idx] > 0.50
    
    # Hyperopic (selected < true)
    bayes_hyperopic[idx] = bayes_power < true_p
    random_hyperopic[idx] = random_power < true_p
    myopic_hyperopic[idx] = myopic_power < true_p
    mle_hyperopic[idx] = mle_power < true_p
    
    # Utility
    bayes_utility[idx] = utility(bayes_power, true_p)
    random_utility[idx] = utility(random_power, true_p)
    myopic_utility[idx] = utility(myopic_power, true_p)
    mle_utility[idx] = utility(mle_power, true_p)

print(f"✅ Decision framework applied to {n:,} triggered cases in {time.time() - start_loop:.2f} seconds\n")

surprise_bayes = np.mean(bayes_surprise) * 100

# ## 6. Clinical Case Example (ESCRS Calculator Split-Screen)

# %% [code]
print("=" * 70)
print("CLINICAL CASE EXAMPLE: ESCRS CALCULATOR SPLIT-SCREEN")
print("=" * 70)

print("\nCase: AL 25.40 mm, K ~42.28 D, Vivity EDOF lens")

formula_outputs = {
    'Cooke_K6': 17.00,
    'EVO': 16.50,
    'Hill_RBF': 17.00,
    'Hoffer_QST': 16.50,
    'Kane': 16.50,
    'Pearl_DGS': 16.50
}

print("\nFormula Outputs:")
for name, power in formula_outputs.items():
    print(f"  {name}: {power:.2f} D")

candidates = np.array([16.50, 17.00])
preds_case = np.array([list(formula_outputs.values())])

# Compute log-likelihoods and posterior
log_likelihoods = []
for cand in candidates:
    log_likelihoods.append(get_log_likelihood(cand, preds_case[0], sigma_values))

log_likelihoods = np.array(log_likelihoods)
max_ll = np.max(log_likelihoods)
likelihoods = np.exp(log_likelihoods - max_ll)
posterior_probs = likelihoods / np.sum(likelihoods)

print(f"\nPosterior probabilities:")
for i, cand in enumerate(candidates):
    print(f"  P(True Power = {cand:.2f} D | Data) = {posterior_probs[i]:.3f}")

# Expected utility for each choice
expected_utilities = []
for j, cand in enumerate(candidates):
    exp_util = 0
    for k, other_cand in enumerate(candidates):
        exp_util += posterior_probs[k] * utility(cand, other_cand)
    expected_utilities.append(exp_util)

print(f"\nExpected utility for each choice:")
for i, cand in enumerate(candidates):
    print(f"  Select {cand:.2f} D: Expected Utility = {expected_utilities[i]:.3f}")

best_idx = np.argmax(expected_utilities)
print(f"\n✅ Bayesian decision framework selects {candidates[best_idx]:.2f} D (maximizes expected utility)")

mle_idx = np.argmax(log_likelihoods)
print(f"   Pure MLE would select {candidates[mle_idx]:.2f} D")

if candidates[best_idx] != candidates[mle_idx]:
    print(f"\n   The Bayesian decision framework diverges from MLE due to asymmetric risk.")
    print(f"   The myopic guard (higher power) is selected to avoid hyperopia.")
else:
    print(f"\n   The Bayesian decision framework agrees with MLE in this case.")

print("\nClinical Takeaway: The Bayesian decision framework balances formula consensus")
print("with asymmetric risk, providing a mathematically justifiable decision that")
print("minimizes the expected clinical cost of hyperopia.\n")
